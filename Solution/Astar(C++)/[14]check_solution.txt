void Puzzle::swap(int &x, int &y) {
   int temp = x;
   x = y;
   y = temp;
}

bool Puzzle::isSolvable(int p[5][5])
{
   int i, j;
   int k = 0;
   int temp[25];
   int inv = 0;

   for (i = 0; i < 5; i++)
   {
      for (j = 0; j < 5; j++)
      {
         temp[k++] = p[i][j];
      }
   }

   for (i = 0; i < 24; i++)
   {
      if (temp[i] != 25)
      {
         for (j = i + 1; j < 25; j++)
         {
            if (temp[j] != 25)
               if (temp[i] > temp[j])
                  inv++;
         }
      }
   }

   if (inv % 2 == 0)
      return 0; // solvable
   else
      return 1; // not solvable

}
